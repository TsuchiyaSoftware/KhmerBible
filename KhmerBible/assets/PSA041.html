៤១
សូម​ប្រោស​ទូលបង្គំ​ឲ្យ​ជា​ឡើង​វិញ​ផង
ទំនុក​នៃ​ស្តេច​ដាវីឌ ដែល​ទ្រង់​លើក​សំរាប់​មេ​ភ្លេង។
១ មាន​ពរ​ហើយ អ្នក​ណា​ដល់​មាន​អធ្យាស្រ័យ
ដល់​ពួក​មនុស្ស​ក្រីក្រ
ព្រះយេហូវ៉ា​ទ្រង់​នឹង​ជួយ​អ្នក​នោះ ឲ្យ​រួច​ក្នុង​ថ្ងៃ​អាក្រក់
២ ព្រះយេហូវ៉ា​ទ្រង់​នឹង​ជួយ​ទំនុក​បំរុង
ហើយ​ថែ​រក្សា​ជីវិត​អ្នក​នោះ
គេ​នឹង​បាន​ពរ​នៅ​ផែនដី​នេះ ហើយ​ទ្រង់​នឹង​មិន​ប្រគល់​គេ
ទៅ​ក្នុង​បំណង​ចិត្ត​នៃ​ពួក​ខ្មាំងសត្រូវ​ឡើយ
៣ ព្រះយេហូវ៉ា​នឹង​ទ្រទ្រាំ​ចំពោះ​គេ​នៅ​គ្រា​ដេក​ឈឺ​នៅ​គ្រែ
ទ្រង់​ក៏​រៀប​គ្រែ​ឲ្យ នៅ​គ្រប់​វេលា​មាន​ធុរៈ។
៤ ៙ ខ្ញុំ​បាន​ទូល​ថា ឱ​ព្រះយេហូវ៉ា​អើយ
សូម​អាណិត​មេត្តា​ដល់​ទូលបង្គំ​ផង
សូម​ប្រោស​ព្រលឹង​ទូលបង្គំ​ឲ្យ​បាន​ជា
ដ្បិត​ទូលបង្គំ​បាន​ធ្វើ​បាប​នឹង​ទ្រង់​ហើយ
៥ ពួក​ខ្មាំងសត្រូវ​នៃ​ទូលបង្គំ
គេ​និយាយ​សេចក្តី​អាក្រក់​ពី​ទូលបង្គំ​ថា
តើ​វេលា​ណា​វា​នឹង​ស្លាប់
ហើយ​ឈ្មោះ​វា​វិនាស​បាត់​ទៅ
៦ បើ​កាល​ណា​គេ​មក​សួរ​ទូលបង្គំ
នោះ​ក៏​ពោល​តែ​សេចក្តី​ភូតភរ​ទទេ
ចិត្ត​គេ​ប្រមូល​សេចក្តី​ទុច្ចរិត​ទុក លុះ​ចេញ​ទៅ​ក្រៅ​ហើយ
នោះ​ក៏​ផ្សាយ​ប្រាប់​សេចក្តី​នោះ
៧ ពួក​អ្នក​ដែល​ស្អប់​ទូលបង្គំ គេ​ខ្សឹបខ្សៀវ​គ្នា
ទាស់​នឹង​ទូលបង្គំ
គេ​បង្កើត​ការ​ប្រទូសរ៉ាយ​នឹង​ទូលបង្គំ។
៨ ៙ គេ​ថា មាន​រោគ​អាក្រក់​ជាប់​នៅ​ខ្លួន​វា
ឥឡូវ​នេះ ដែល​វា​ដេក​ចុះ​ហើយ
មុខ​ជា​នឹង​ក្រោក​ឡើង​វិញ​មិន​បាន​ឡើយ
៩ អើ ទាំង​មិត្រ​សំឡាញ់​ស្និទ្ធស្នាល​របស់​ទូលបង្គំ
ជា​អ្នក​ដែល​ទូលបង្គំ​បាន​ទុក​ចិត្ត
ហើយ​ក៏​បាន​បរិភោគ​បាយ​របស់​ទូលបង្គំ​ផង
នោះ​បាន​លើក​កែងជើង​ចង់​ជាន់​ឈ្លី​ទូលបង្គំ​ដែរ
១០ តែ ឱ​ព្រះយេហូវ៉ា​អើយ សូម​ទ្រង់​អាណិត
មេត្តា​លើក​ទូលបង្គំ​ឡើង​វិញ
ដើម្បី​ឲ្យ​ទូលបង្គំ​បាន​សង​តប​ដល់​គេ។
១១ ៙ យ៉ាង​នោះ ទូលបង្គំ​នឹង​ដឹង​ថា
ទ្រង់​សព្វ​ព្រះហឫទ័យ​នឹង​ទូលបង្គំ​ពិត
ដោយ​ពួក​សត្រូវ​មិន​មាន​ជ័យជំនះ​ឈ្នះ​ទូលបង្គំ​ឡើយ
១២ ឯ​ទូលបង្គំ ទ្រង់​ទប់ទល់
ដោយ​សេចក្តី​សុចរិត​របស់​ទូលបង្គំ
ហើយ​ក៏​តាំង​ទូលបង្គំ
ឲ្យ​នៅ​ចំពោះ​ទ្រង់​ជា​ដរាប​ត​ទៅ។
១៣ ៙ សូម​លើក​សរសើរ​ដល់​ព្រះយេហូវ៉ា
ជា​ព្រះ​នៃ​សាសន៍​អ៊ីស្រាអែល
ចាប់​តាំង​ពី​អតីត ទៅ​ដល់​អនាគត អស់កល្ប​ជានិច្ច
អាម៉ែន ហើយ​អាម៉ែន។
